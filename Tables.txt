Tables

1.User registration for login
CREATE TABLE USER
(
	USERID int NOT NULL AUTO_INCREMENT ,
	USERNAME VARCHAR(60),CONSTRAINT USERNAME_UNIQUE UNIQUE (USERNAME),
	PASSWORD VARCHAR(50),
	MAIL VARCHAR(30),
	PHONE VARCHAR(15),
	primary KEY(USERID)
);

2.Admission Enquiry

CREATE TABLE ADMINENQ
(
	ADMINENQID INT NOT NULL AUTO_INCREMENT,
	ENQFOR varchar(20),
	KIDNAME VARCHAR(60),
	KIDDOB DATETIME,
	KIDAGE NUMERIC(3,0),
	FATHERNAME VARCHAR(60),
	FATHERMOB NUMERIC(12,0),
	FATHEROCC VARCHAR(30),
	MOTHERNAME VARCHAR(60),
	MOTHERMOB NUMERIC(12,0),
	MOTHEROCC VARCHAR(30),
	RESADDRESS TEXT,
	LANDMARK VARCHAR(40),
	HOWDOKNOW VARCHAR(40),
	REFERENCE VARCHAR(30),
	FEETOLD NUMERIC(18,4),
	OFFERGIVEN VARCHAR(15),
	FINALFEE NUMERIC(18,4),
	TYPEOFENQUIRY VARCHAR(18),
	STATUS VARCHAR(10),
	FINALSTATUS VARCHAR(20),
	FINALDATE DATETIME,
	FOLLOWUP1 VARCHAR(20),
	FOLL1DATE DATETIME,
	FOLLOWUP2 VARCHAR(20),
	FOLL2DATE DATETIME,
	FOLLOWUP3 VARCHAR(20),
	FOLL3DATE DATETIME,

);
